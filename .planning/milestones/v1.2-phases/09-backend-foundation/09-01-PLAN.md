```python
"""Pydantic models for GoHighLevel API integration."""
from __future__ import annotations

from datetime import datetime
from typing import Optional
from pydantic import BaseModel, Field


class GHLConnectionCreate(BaseModel):
    """Request model for creating a GHL connection."""
    name: str = Field(..., min_length=1, max_length=100, description="Connection display name")
    token: str = Field(..., min_length=10, description="Private Integration Token")
    location_id: str = Field(..., min_length=1, description="GHL Location ID")
    notes: Optional[str] = Field(None, max_length=500, description="Optional notes")


class GHLConnectionUpdate(BaseModel):
    """Request model for updating a GHL connection. Token is optional (only set if changing)."""
    name: Optional[str] = Field(None, min_length=1, max_length=100)
    token: Optional[str] = Field(None, min_length=10, description="New token (omit to keep existing)")
    location_id: Optional[str] = Field(None, min_length=1)
    notes: Optional[str] = Field(None, max_length=500)


class GHLConnectionResponse(BaseModel):
    """Response model for GHL connection (never includes encrypted_token)."""
    id: str
    name: str
    token_last4: str = Field(description="Last 4 chars of token for masked display")
    location_id: str
    notes: str = ""
    validation_status: str = Field(description="pending | valid | invalid")
    created_at: datetime
    updated_at: datetime


class ContactUpsertRequest(BaseModel):
    """Request model for upserting a single contact to GHL."""
    connection_id: str = Field(..., description="GHL connection ID to use")
    first_name: Optional[str] = None
    last_name: Optional[str] = None
    email: Optional[str] = None
    phone: Optional[str] = None
    address1: Optional[str] = None
    city: Optional[str] = None
    state: Optional[str] = None
    postal_code: Optional[str] = None
    tags: list[str] = Field(default_factory=list)
    assigned_to: Optional[str] = Field(None, description="GHL user ID for contact owner")


class ContactUpsertResponse(BaseModel):
    """Response model for contact upsert result."""
    success: bool
    action: str = Field(description="created | updated | failed")
    ghl_contact_id: Optional[str] = None
    error: Optional[str] = None


class GHLUserResponse(BaseModel):
    """GHL user from /users/ endpoint, used for contact owner dropdown."""
    id: str
    name: str
    email: str
    role: Optional[str] = None


class GHLValidationResult(BaseModel):
    """Result of validating a GHL connection."""
    valid: bool
    error: Optional[str] = None
    location_name: Optional[str] = None
    users: list[GHLUserResponse] = Field(default_factory=list)
```

2. Create `toolbox/backend/app/services/ghl/normalization.py`:
   - `normalize_phone(phone: str) -> Optional[str]`: Parse with `phonenumbers.parse(phone, "US")`, validate, format to E.164. Return None if invalid/empty.
   - `normalize_email(email: str) -> Optional[str]`: Strip whitespace, lowercase, validate has `@` and domain with `.`. Return None if invalid/empty.
   - `normalize_name(name: str) -> str`: Strip whitespace, apply `.title()` as safety net. Return empty string if None/empty.
   - `normalize_contact(data: dict) -> dict`: Apply all normalizations to a contact dict. Keys: first_name, last_name, email, phone. Returns normalized dict.
   - `validate_contact(data: dict) -> tuple[bool, Optional[str]]`: Check contact has at least email OR phone after normalization. Return (True, None) if valid, (False, "error message") if invalid.
   - Use `logger = logging.getLogger(__name__)` for warnings on invalid phone/email.
   - Never log actual phone/email values — only log "Invalid phone format" not "Invalid phone: 555-1234".

3. Create `toolbox/backend/app/services/ghl/client.py`:
   - `RateLimiter` class: Token bucket pattern. `__init__(max_requests=50, period_seconds=10.0)`. `async acquire()` method that waits if tokens exhausted, refills after period.
   - `GHLClient` class: Async context manager (`__aenter__`/`__aexit__`).
     - `BASE_URL = "https://services.leadconnectorhq.com"`
     - `VERSION = "2021-07-28"` (GHL API version header)
     - `__init__(self, token: str, location_id: str)`: Store token and location_id, create RateLimiter.
     - `__aenter__`: Create `httpx.AsyncClient` with base_url, Authorization Bearer header, Version header, Content-Type application/json, timeout 30s (connect 10s).
     - `__aexit__`: Close httpx client.
     - `async _request(method, endpoint, max_retries=3, **kwargs) -> dict`:
       - Acquire rate limit token.
       - Make request. On 429: exponential backoff (2^attempt seconds), retry up to max_retries.
       - On success: log one-line summary (`logger.info(f"{method} {endpoint} -> {status_code}")`). Return response.json().
       - On HTTPStatusError: log error with status code and response text (no tokens/PII). Re-raise.
       - On RequestError: log error. Re-raise.
       - After max retries exhausted: raise `GHLRateLimitError`.
     - `async get_users() -> dict`: `GET /users/` with params `{"locationId": self.location_id}`. Used for token validation and contact owner dropdown.
     - `async search_contacts(email: str) -> list[dict]`: `GET /contacts/` with params `{"email": email, "locationId": self.location_id}`. Returns list of matching contacts.
     - `async create_contact(contact_data: dict) -> dict`: `POST /contacts/` with json body. Body must include `locationId`.
     - `async update_contact(contact_id: str, contact_data: dict) -> dict`: `PUT /contacts/{contact_id}` with json body.
     - `async upsert_contact(contact_data: dict) -> dict`:
       - Import and call `normalize_contact` and `validate_contact` from normalization module.
       - If validation fails, raise ValueError with error message.
       - Map our field names to GHL field names: first_name->firstName, last_name->lastName, email->email, phone->phone, address1->address1, city->city, state->state, postal_code->postalCode.
       - Add `locationId` from self.location_id.
       - Add tags if present.
       - Add `assignedTo` if assigned_to is set.
       - Search by email first. If found, update. If not found, create.
       - Return dict with keys: action ("created"|"updated"), contact (GHL response), ghl_contact_id.
   - Define custom exceptions at module level:
     - `class GHLAPIError(Exception): pass`
     - `class GHLRateLimitError(GHLAPIError): pass`
     - `class GHLAuthError(GHLAPIError): pass`

   Key implementation notes:
   - Use `from __future__ import annotations` at top.
   - `logger = logging.getLogger(__name__)` per module convention.
   - NEVER log tokens or PII (phone, email). Only log method, endpoint, status code.
   - Pass through GHL error details to caller per user decision (don't normalize to generic errors).
   - In _request, on non-429 HTTPStatusError, check if status is 401 and raise GHLAuthError instead.
